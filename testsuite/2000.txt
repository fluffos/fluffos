def _make_2000_lines():
  import random
  r = random.Random(100)
  chars = list(map(chr, range(32, 128)))
  i = 0
  while i < len(chars):
    chars[i] = " "
    i += r.randrange(9)
  s = "".join(chars)
  s = s*4
  words = []
  for i in range(2000):
    start = r.randrange(96)
    n = r.randint(5, 65)
    words.append(s[start:start+n])
  return "\n".join(words)+"\n"
